交换两个整型变量的值：

```c
#include<stdio.h>
 int main()
 {
     int a = 3;
     int b = 5;
     //整型溢出
     a = a + b;
     b = a - b;
     a = a - b;
     printf("交换后：a=%d,b=%d",a,b);
     //按位异或 - 效率低
     a = a^b;
     b = a^b;
     a = a^b;
     printf("交换后：a=%d,b=%d",a,b);
 }
```

找出单身狗

```c
#include<stdio.h>
 int main();
 {
     int arr[] = {1,2,3,4,5,4,2,3,1,5,7};
     //暴力求解
     int i = 0;
     int sz = sizeof(arr) / sizeof(arr[0]);
     for (i = 0;i<sz;i++)
     {
         int count = 0;
         int j = 0;
         for (j = 0;j<sz;j++)
         {
             if (arr[i]==arr[j])
                 count++;
         }
         if (count==1)
         {
             printf("单身狗：%d\n",i);
             break;
         }
     }
     //异或妙想
     int ret = 0;
     for (i = 0;i < sz;i++)
     {
         ret = ret^arr[i];
     }
     printf("单身狗：%d\n",ret);
     return 0;
 }
```
